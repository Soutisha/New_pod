# test_app.py

import pytest
from app import app

@pytest.fixture
def client():
    app.config["TESTING"] = True
    with app.test_client() as c:
        yield c

def test_index(client):
    response = client.get('/')
    assert response.status_code == 200
    assert response.content_type == 'text/html'

def test_register_get(client):
    response = client.get('/register')
    assert response.status_code == 200
    assert response.content_type == 'text/html'

def test_register_post(client):
    response = client.post('/register', data={'username': 'testuser', 'password': 'testpassword'})
    assert response.status_code == 302
    assert response.location == 'http://localhost/login'

def test_login_get(client):
    response = client.get('/login')
    assert response.status_code == 200
    assert response.content_type == 'text/html'

def test_login_post(client):
    client.post('/register', data={'username': 'testuser', 'password': 'testpassword'})
    response = client.post('/login', data={'username': 'testuser', 'password': 'testpassword'})
    assert response.status_code == 302
    assert response.location == 'http://localhost/dashboard'

def test_dashboard(client):
    client.post('/register', data={'username': 'testuser', 'password': 'testpassword'})
    client.post('/login', data={'username': 'testuser', 'password': 'testpassword'})
    response = client.get('/dashboard')
    assert response.status_code == 200
    assert response.content_type == 'text/html'

def test_logout(client):
    client.post('/register', data={'username': 'testuser', 'password': 'testpassword'})
    client.post('/login', data={'username': 'testuser', 'password': 'testpassword'})
    response = client.get('/logout')
    assert response.status_code == 302
    assert response.location == 'http://localhost/'

def test_attendance(client):
    client.post('/register', data={'username': 'testuser', 'password': 'testpassword'})
    client.post('/login', data={'username': 'testuser', 'password': 'testpassword'})
    response = client.post('/attendance', data={'date': '2022-01-01', 'attended': 'true'})
    assert response.status_code == 302
    assert response.location == 'http://localhost/dashboard'

def test_attendance_invalid_date(client):
    client.post('/register', data={'username': 'testuser', 'password': 'testpassword'})
    client.post('/login', data={'username': 'testuser', 'password': 'testpassword'})
    response = client.post('/attendance', data={'date': 'invalid-date', 'attended': 'true'})
    assert response.status_code == 302
    assert response.location == 'http://localhost/dashboard'

def test_attendance_invalid_attended(client):
    client.post('/register', data={'username': 'testuser', 'password': 'testpassword'})
    client.post('/login', data={'username': 'testuser', 'password': 'testpassword'})
    response = client.post('/attendance', data={'date': '2022-01-01', 'attended': 'invalid'})
    assert response.status_code == 302
    assert response.location == 'http://localhost/dashboard'

def test_404(client):
    response = client.get('/non-existent-route')
    assert response.status_code == 404
    assert response.content_type == 'text/html'

def test_empty_register(client):
    response = client.post('/register', data={'username': '', 'password': ''})
    assert response.status_code == 200
    assert response.content_type == 'text/html'
    assert 'Username cannot be empty' in response.data.decode('utf-8')
    assert 'Password cannot be empty' in response.data.decode('utf-8')

def test_empty_login(client):
    response = client.post('/login', data={'username': '', 'password': ''})
    assert response.status_code == 200
    assert response.content_type == 'text/html'
    assert 'Username cannot be empty' in response.data.decode('utf-8')
    assert 'Password cannot be empty' in response.data.decode('utf-8')
