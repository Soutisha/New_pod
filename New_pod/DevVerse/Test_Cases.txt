def test_database(client):
    """Test table creation and model CRUD"""
    attendance = Attendance(name="test", date="2022-01-01", status="present")
    db.session.add(attendance)
    db.session.commit()
    assert Attendance.query.count() == 1
    attendance = Attendance.query.first()
    assert attendance.name == "test"
    assert attendance.date == "2022-01-01"
    assert attendance.status == "present"
    db.session.delete(attendance)
    db.session.commit()
    assert Attendance.query.count() == 0

def test_database_uniqueness(client):
    """Test uniqueness constraints"""
    attendance1 = Attendance(name="test", date="2022-01-01", status="present")
    db.session.add(attendance1)
    db.session.commit()
    attendance2 = Attendance(name="test", date="2022-01-01", status="present")
    db.session.add(attendance2)
    with pytest.raises(db.IntegrityError):
        db.session.commit()
