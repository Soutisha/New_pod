def test_edge_cases_security(client):
    """Test SQL injection attempt, XSS in input, empty fields, very long input (10000 chars), missing fields"""
    # SQL injection attempt
    response = client.post("/login", data={"username": "test_user", "password": " OR 1=1"})
    assert response.status_code == 200
    # XSS in input
    response = client.post("/register", data={"username": "<script>alert('XSS')</script>", "email": "test@example.com", "password": "password"})
    assert response.status_code == 302
    # Empty fields
    response = client.post("/register", data={"username": "", "email": "", "password": ""})
    assert response.status_code == 200
    # Very long input (10000 chars)
    response = client.post("/register", data={"username": "a" * 10000, "email": "test@example.com", "password": "password"})
    assert response.status_code == 200
    # Missing fields
    response = client.post("/register", data={"username": "test_user", "password": "password"})
    assert response.status_code == 200
